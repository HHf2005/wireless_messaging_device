// 3. CRYPTO FUNCTIONS
// =============================================================
void updateAesKeyArray() {
  String fullString = FIXED_KEY_PART + userPin; 
  if (fullString.length() != 16) {
      while(fullString.length() < 16) fullString += "0";
      if(fullString.length() > 16) fullString = fullString.substring(0, 16);
  }
  for(int i=0; i<16; i++) aes_key[i] = (byte)fullString[i];
}

void aes_init() { aesLib.set_paddingmode(paddingMode::CMS); }

String encrypt(String msg) {
  aes_init();
  int len = msg.length();
  int outLen = aesLib.get_cipher64_length(len);
  char outBuf[outLen + 10];
  byte t_iv[16]; memcpy(t_iv, aes_iv, 16);
  aesLib.encrypt64((byte*)msg.c_str(), len, outBuf, aes_key, 128, t_iv);
  return String(outBuf);
}

String decrypt(String msg) {
  aes_init();
  int len = msg.length();
  byte outBuf[len + 20]; memset(outBuf, 0, sizeof(outBuf));
  byte t_iv[16]; memcpy(t_iv, aes_iv, 16);
  aesLib.decrypt64((char*)msg.c_str(), len, outBuf, aes_key, 128, t_iv);
  String cleanText = "";
  for (int i = 0; i < sizeof(outBuf); i++) {
    char c = (char)outBuf[i];
    if (c == 0) break;
    if (isPrintable(c)) cleanText += c;
  }
  return cleanText;
}

void saveSettings() {
  preferences.begin("messenger", false); 
  preferences.putBool("sound", isSoundEnabled);
  preferences.putString("user_pin", userPin); 
  preferences.end(); 
  updateAesKeyArray(); 
}

void loadSettings() {
  preferences.begin("messenger", true); 
  isSoundEnabled = preferences.getBool("sound", true); 
  userPin = preferences.getString("user_pin", "0000"); 
  preferences.end();
  updateAesKeyArray(); 
}

void playTone() {
  if (isSoundEnabled) { 
    digitalWrite(BUZZER_PIN, HIGH); delay(150); digitalWrite(BUZZER_PIN, LOW); 
  }
}

void showNotification(String text, uint16_t color, int y) {
  tft.fillRect(0, y - 5, 320, 25, ILI9341_BLACK); 
  tft.setCursor(10, y); tft.setTextColor(color); tft.setTextSize(2); tft.print(text);
}

void drawChatLog() {
  tft.fillRect(0, 10, 320, tft.height() - 60, ILI9341_BLACK); 
  int currentY = 10; 
  int endIndex = min(messageCount, scrollOffset + MESSAGES_ON_SCREEN); 
  for (int i = scrollOffset; i < endIndex; i++) {
    tft.setCursor(5, currentY); 
    if (chatLog[i].startsWith("\x01")) tft.setTextColor(ILI9341_GREEN, ILI9341_BLACK); 
    else tft.setTextColor(ILI9341_YELLOW, ILI9341_BLACK); 
    tft.setTextSize(2);
    tft.print(chatLog[i].substring(1)); 
    currentY += 20; 
  }
}

void drawInputText() {
    int TEXT_Y = tft.height() - 25; 
    tft.fillRect(15, TEXT_Y - 2, 300, 22, ILI9341_GRAY); 
    tft.setCursor(15, TEXT_Y); tft.setTextColor(ILI9341_WHITE); tft.setTextSize(2);
    String before = inputText.substring(0, cursorPosition);
    String after = inputText.substring(cursorPosition);
    if (cursorVisible) before += "_"; 
    tft.print(before + after);
}

void drawCapsLockIndicator() {
    int y = tft.height() - 45 - 25; 
    tft.fillRect(300, y, 20, 20, ILI9341_BLACK);
    tft.setCursor(300, y); tft.setTextSize(2);
    if (isCapsLock) { tft.setTextColor(ILI9341_RED); tft.print("A"); } 
    else { tft.setTextColor(ILI9341_GREEN); tft.print("a"); }

    drawCapsLockIndicator(); drawInputText(); drawChatLog();   
}

    else { tft.setTextColor(ILI9341_RED, bgSnd); tft.print("OFF"); }
}

void addToChatLog(String message, bool isSent) {
  String prefix = isSent ? "\x01" : "\x02";
  String formattedMessage = prefix + message;
  if (messageCount < MAX_MESSAGES) chatLog[messageCount++] = formattedMessage;
  else {
      for (int i = 0; i < MAX_MESSAGES - 1; i++) chatLog[i] = chatLog[i + 1]; 
      chatLog[MAX_MESSAGES - 1] = formattedMessage; 
  }
  if (messageCount > MESSAGES_ON_SCREEN) scrollOffset = messageCount - MESSAGES_ON_SCREEN; 
  else scrollOffset = 0; 
  if (currentMode == CHAT_MODE) drawChatLog(); 
}

// 6. INPUT HANDLING
// =============================================================
void handleSettingsMode(char key) {
    if (key == '!') { isEditingPin = false; currentMode = CHAT_MODE; drawChatUI(); return; }

    if (isEditingPin) {
        if (key == '&' || key == '*') { isEditingPin = false; drawSettingsUI(); return; } 
        if (key >= '0' && key <= '9' && tempPinInput.length() < 4) { tempPinInput += key; drawSettingsUI(); }
        else if (key == '-') { if (tempPinInput.length() > 0) tempPinInput.remove(tempPinInput.length() - 1); drawSettingsUI(); }
        else if (key == '@') {
            if (tempPinInput.length() > 0) {
                while(tempPinInput.length() < 4) tempPinInput += "0";
                userPin = tempPinInput; saveSettings(); isEditingPin = false; 
                // 🔥 تم الإصلاح هنا بإضافة 220
                showNotification("Saved!", ILI9341_GREEN, 220);
            } else { isEditingPin = false; }
            drawSettingsUI();
        }
        return; 
    }

    if (key == '&') { menuIndex--; if (menuIndex < 0) menuIndex = MENU_ITEMS - 1; drawSettingsUI(); }
    else if (key == '*') { menuIndex++; if (menuIndex >= MENU_ITEMS) menuIndex = 0; drawSettingsUI(); }
    else if (key == '@') {
        if (menuIndex == 0) { isEditingPin = true; tempPinInput = ""; drawSettingsUI(); }
        else if (menuIndex == 1) { isSoundEnabled = !isSoundEnabled; saveSettings(); drawSettingsUI(); }
    }
}

void handleChatMode(char key) {
  if (key == '!') { currentMode = SETTINGS_MODE; drawSettingsUI(); return; }

  if (key == '@') { 
    if (inputText.length() > 0) {
      String encryptedText = encrypt(inputText);
      Serial.print("Enc: "); Serial.println(encryptedText);
      LoRa.beginPacket(); LoRa.print(encryptedText); LoRa.endPacket();
      addToChatLog(inputText, true); 
      inputText = ""; cursorPosition = 0; drawInputText();
    }
    return;
  }
  
  if (key == '^' || key == '%' || key == '$' || key == '/') { 
      if (inputText.length() < MAX_INPUT_LENGTH) {
        inputText = inputText.substring(0, cursorPosition) + " " + inputText.substring(cursorPosition);
        cursorPosition++; drawInputText();
      }
      return;
  }
  
  if (key == '.') { isCapsLock = !isCapsLock; drawCapsLockIndicator(); return; }
  if (key == '>') { if (cursorPosition < inputText.length()) { cursorPosition++; drawInputText(); } return; }
  if (key == '<') { if (cursorPosition > 0) { cursorPosition--; drawInputText(); } return; }
  if (key == '&') { if (scrollOffset > 0) { scrollOffset--; drawChatLog(); } return; } 
  if (key == '*') { int maxScroll = messageCount - MESSAGES_ON_SCREEN; if (maxScroll < 0) maxScroll = 0; if (scrollOffset < maxScroll) { scrollOffset++; drawChatLog(); } return; }
  if (key == '-') { if (cursorPosition > 0) { inputText = inputText.substring(0, cursorPosition - 1) + inputText.substring(cursorPosition); cursorPosition--; drawInputText(); } return; }
  if (key == '#') { inputText = ""; cursorPosition = 0; drawInputText(); return; }

  if (inputText.length() < MAX_INPUT_LENGTH) { 
    String charToAdd = String(key);
    if (key >= 'A' && key <= 'Z' && !isCapsLock) charToAdd.toLowerCase();
    else if (key == 'k' && isCapsLock) charToAdd = "K";
    inputText = inputText.substring(0, cursorPosition) + charToAdd + inputText.substring(cursorPosition);
    cursorPosition++; drawInputText();
  }
}

// 7. MAIN SETUP & LOOP
// =============================================================
void setup() {
  Serial.begin(115200); 
  Wire.begin(); 
  pinMode(BUZZER_PIN, OUTPUT); digitalWrite(BUZZER_PIN, LOW); 
  loadSettings(); 
  
  if (!aw.begin(0x58)) { Serial.println("AW9523R Fail!"); while (1); }
  for (int r = 0; r < ROWS; r++) { aw.pinMode(rowPins[r], OUTPUT); aw.digitalWrite(rowPins[r], HIGH); } 
  for (int c = 0; c < COLS; c++) { aw.pinMode(colPins[c], INPUT_PULLUP); } 

  SPI.begin(LORA_SCK, LORA_MISO, LORA_MOSI, LORA_CS);
  LoRa.setPins(LORA_CS, LORA_RST, LORA_DIO0);

  tft.begin(); tft.setRotation(3); 
  
  showBootScreen();

  if (LoRa.begin(433E6)) {
      LoRa.setTxPower(20);
      LoRa.setSpreadingFactor(12);
      LoRa.setSignalBandwidth(62.5E3);
      LoRa.setCodingRate4(8);
      Serial.println("LoRa OK!"); 
  } else { 
      tft.setTextColor(ILI9341_RED); tft.println("LoRa Fail!"); while (1); 
  } 
  
  drawChatUI(); 
}

void loop() {
  if (currentMode == CHAT_MODE) {
      if (millis() - lastCursorToggle >= CURSOR_INTERVAL) {
        lastCursorToggle = millis(); cursorVisible = !cursorVisible; drawInputText(); 
      }
  }

  char key = '\0'; 
  for (int r = 0; r < ROWS; r++) {
    aw.digitalWrite(rowPins[r], LOW); delayMicroseconds(50); 
    for (int c = 0; c < COLS; c++) {
      if (aw.digitalRead(colPins[c]) == LOW) {
        if (!keyState[r][c]) { keyState[r][c] = true; key = keys[r][c]; cursorVisible = true; lastCursorToggle = millis(); }
      } else keyState[r][c] = false; 
    }
    aw.digitalWrite(rowPins[r], HIGH); if (key != '\0') break;
  }
  
  if (key != '\0') {
      if (currentMode == SETTINGS_MODE && key == '!') { currentMode = CHAT_MODE; drawChatUI(); }
      else if (currentMode == CHAT_MODE) handleChatMode(key);
      else handleSettingsMode(key);
  }
  
  int packetSize = LoRa.parsePacket(); 
  if (packetSize) {
    String receivedEncrypted = "";
    while (LoRa.available()) receivedEncrypted += (char)LoRa.read();
    Serial.print("Enc: "); Serial.println(receivedEncrypted);
    String decryptedText = decrypt(receivedEncrypted);
    addToChatLog(decryptedText, false); playTone(); 
  }
}
