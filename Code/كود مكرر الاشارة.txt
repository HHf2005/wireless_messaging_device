#include <SPI.h>
#include <LoRa.h>

#define LORA_SCK   18
#define LORA_MISO  19
#define LORA_MOSI  23
#define LORA_CS    5
#define LORA_RST   17
#define LORA_DIO0  16

void setup() {
  Serial.begin(115200);
  
  SPI.begin(LORA_SCK, LORA_MISO, LORA_MOSI, LORA_CS);
  LoRa.setPins(LORA_CS, LORA_RST, LORA_DIO0);

  if (!LoRa.begin(433E6)) {
    Serial.println("Repeater Init Failed!");
    while (1);
  }

  // نفس إعدادات المدى البعيد
  LoRa.setTxPower(20);
  LoRa.setSpreadingFactor(12);
  LoRa.setSignalBandwidth(62.5E3);
  LoRa.setCodingRate4(8);
  
  Serial.println("--- LORA REPEATER ACTIVE ---");
}

void loop() {
  int packetSize = LoRa.parsePacket();
  
  if (packetSize) {
    // 1. استقبال الرسالة
    String incomingMsg = "";
    while (LoRa.available()) {
      incomingMsg += (char)LoRa.read();
    }
    
    Serial.print("Relaying packet: ");
    Serial.println(incomingMsg);

    delay(200);

    LoRa.beginPacket();
    LoRa.print(incomingMsg);
    LoRa.endPacket();
    
    Serial.println(" -> Re-transmitted!");
  }
}
